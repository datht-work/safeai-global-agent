# Global Compliance Standards & Certifications

<law_definition id="ISO-27001">
  <name>ISO/IEC 27001:2022 — Information Security Management</name>
  <effective_date>2022-10-25</effective_date>

  <rule category="security_controls">
    <description>
      Requires a risk-based approach to security. Mandatory controls include Access Control (A.5.15), Crypto (A.8.24), and Logging (A.8.15).
    </description>
  </rule>
</law_definition>

<law_definition id="ISO-42001">
  <name>ISO/IEC 42001:2023 — Artificial Intelligence Management System</name>
  <effective_date>2023-12-18</effective_date>

  <rule category="ai_governance">
    <description>
      Defines organizational roles for AI, risk assessments, and transparency requirements throughout the AI lifecycle.
    </description>
  </rule>
</law_definition>

<law_definition id="SOC2">
  <name>SOC 2 — Trust Services Criteria</name>
  <rule category="security_controls">
    <description>
      Maps to Security, Availability, Processing Integrity, Confidentiality, and Privacy criteria.
    </description>
  </rule>
</law_definition>

<law_definition id="WCAG-2.2">
  <name>Web Content Accessibility Guidelines (WCAG) 2.2</name>
  <rule category="accessibility">
    <description>
      Requires digital content to be Perceivable, Operable, Understandable, and Robust (POUR).
    </description>
  </rule>
</law_definition>
